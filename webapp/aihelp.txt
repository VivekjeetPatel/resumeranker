You are building a full-stack AI-powered Resume Ranker web application using Python. Here is the complete specification:

---

## PROJECT OVERVIEW

A web app where a recruiter can:
1. Paste a job description
2. Set how many top resumes to shortlist (e.g. top 10, 20, 50)
3. Upload a ZIP file containing resumes (PDF, DOC, DOCX)
4. Click "Run AI Ranking" — the system processes all resumes through Gemini AI and returns a ranked table
5. Download shortlisted resumes as a ZIP or export results as CSV

---

## TECH STACK

- **Backend**: Python, FastAPI
- **Frontend**: Jinja2 templates + Tailwind CSS (no separate frontend framework)
- **Resume Parsing**: pdfplumber (PDF), python-docx (DOCX), mammoth (DOC)
- **AI**: Google Gemini API (gemini-1.5-flash) via google-generativeai SDK
- **File handling**: Python zipfile module
- **Database**: SQLite via SQLAlchemy
- **Environment**: python-dotenv (.env file with GEMINI_API_KEY)

## STEP 1 — core/jd_extractor.py

Use Gemini to extract structured requirements from raw JD text.

```python
import google.generativeai as genai
import json, os
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel("gemini-1.5-flash")

def extract_jd_requirements(jd_text: str) -> dict:
    prompt = f"""
You are an expert HR analyst. Extract the key requirements from this job description.

Job Description:
{jd_text}

Return ONLY a valid JSON object with this exact structure:
{{
  "role": "Job title",
  "required_skills": ["skill1", "skill2"],
  "preferred_skills": ["skill1", "skill2"],
  "experience_years": 3,
  "education": "Bachelor's in Computer Science or related field",
  "responsibilities": ["responsibility1", "responsibility2"],
  "keywords": ["keyword1", "keyword2"]
}}
Return only JSON, no explanation, no markdown.
"""
    response = model.generate_content(prompt)
    text = response.text.strip().strip("```json").strip("```").strip()
    return json.loads(text)
```

## STEP 2 — core/scorer.py

For EACH resume, call Gemini with both JD requirements and resume text. Return structured score.

```python
import google.generativeai as genai
import json, os, time
from dotenv import load_dotenv

load_dotenv()
genai.configure(api_key=os.getenv("GEMINI_API_KEY"))
model = genai.GenerativeModel("gemini-1.5-flash")

def score_resume(jd_requirements: dict, resume_text: str, candidate_name: str) -> dict:
    prompt = f"""
You are an expert technical recruiter. Score this resume against the job requirements.

JOB REQUIREMENTS:
{json.dumps(jd_requirements, indent=2)}

RESUME:
{resume_text[:4000]}

Score this candidate and return ONLY a valid JSON object with this exact structure:
{{
  "candidate": "{candidate_name}",
  "score": <overall fit score 0-100>,
  "skills_match": <skills match score 0-100>,
  "experience_match": <experience match score 0-100>,
  "education_match": <education match score 0-100>,
  "missing_skills": ["skill1", "skill2"],
  "strengths": ["strength1", "strength2", "strength3"],
  "summary": "One sentence summary of this candidate's fit"
}}

Scoring rules:
- score: weighted average (skills 40%, experience 40%, education 20%)
- skills_match: percentage of required_skills found in resume
- experience_match: how well years/depth of experience matches
- education_match: how well education background matches
- missing_skills: required skills NOT found in resume
- strengths: top 3-5 things this candidate does well for THIS role
- Be strict and realistic. Do not inflate scores.

Return only JSON, no explanation, no markdown.
"""
    try:
        response = model.generate_content(prompt)
        text = response.text.strip().strip("```json").strip("```").strip()
        result = json.loads(text)
        result["candidate"] = candidate_name  # always override with actual filename
        return result
    except Exception as e:
        return {
            "candidate": candidate_name,
            "score": 0,
            "skills_match": 0,
            "experience_match": 0,
            "education_match": 0,
            "missing_skills": [],
            "strengths": [],
            "summary": f"Could not parse resume: {str(e)}"
        }

def score_all_resumes(jd_requirements: dict, resumes: list[dict]) -> list[dict]:
    """
    resumes: list of {filename, text}
    Returns list of score dicts sorted by score descending
    """
    results = []
    for i, resume in enumerate(resumes):
        candidate_name = resume["filename"].replace("_", " ").replace("-", " ").rsplit(".", 1)[0].title()
        print(f"Scoring {i+1}/{len(resumes)}: {candidate_name}")
        score = score_resume(jd_requirements, resume["text"], candidate_name)
        score["filename"] = resume["filename"]
        score["file_path"] = resume["file_path"]
        results.append(score)
        time.sleep(0.5)  # avoid Gemini rate limiting
    results.sort(key=lambda x: x["score"], reverse=True)
    return results
```

## STEP 3 — core/parser.py

```python
import pdfplumber
from docx import Document
import mammoth
import os

def extract_text_from_pdf(path: str) -> str:
    try:
        with pdfplumber.open(path) as pdf:
            return "\n".join(page.extract_text() or "" for page in pdf.pages)
    except:
        return ""

def extract_text_from_docx(path: str) -> str:
    try:
        doc = Document(path)
        return "\n".join(p.text for p in doc.paragraphs)
    except:
        return ""

def extract_text_from_doc(path: str) -> str:
    try:
        with open(path, "rb") as f:
            result = mammoth.extract_raw_text(f)
            return result.value
    except:
        return ""

def parse_resume(path: str) -> dict:
    ext = os.path.splitext(path)[1].lower()
    if ext == ".pdf":
        text = extract_text_from_pdf(path)
    elif ext == ".docx":
        text = extract_text_from_docx(path)
    elif ext == ".doc":
        text = extract_text_from_doc(path)
    else:
        return {"filename": os.path.basename(path), "text": "", "file_path": path, "error": "Unsupported format"}
    return {
        "filename": os.path.basename(path),
        "text": text.strip(),
        "file_path": path,
        "error": None if text.strip() else "Could not extract text"
    }

def parse_all_resumes(folder_path: str) -> list[dict]:
    results = []
    for root, dirs, files in os.walk(folder_path):
        for fname in files:
            if fname.lower().endswith((".pdf", ".doc", ".docx")):
                full_path = os.path.join(root, fname)
                results.append(parse_resume(full_path))
    return [r for r in results if r["text"]]  # skip unreadable
```

## DATABASE SCHEMA — database.py

```python
from sqlalchemy import create_engine, Column, String, Integer, Float, Boolean, DateTime, Text
from sqlalchemy.ext.declarative import declarative_base
from sqlalchemy.orm import sessionmaker
import datetime

SQLALCHEMY_DATABASE_URL = "sqlite:///./resume_ranker.db"
engine = create_engine(SQLALCHEMY_DATABASE_URL, connect_args={"check_same_thread": False})
SessionLocal = sessionmaker(bind=engine)
Base = declarative_base()

class JobRun(Base):
    __tablename__ = "job_runs"
    id = Column(String, primary_key=True)
    created_at = Column(DateTime, default=datetime.datetime.utcnow)
    job_description = Column(Text)
    jd_requirements = Column(Text)   # JSON string
    shortlist_count = Column(Integer)
    total_resumes = Column(Integer, default=0)
    processed_resumes = Column(Integer, default=0)
    time_taken_seconds = Column(Float, default=0)
    status = Column(String, default="pending")  # pending, processing, done, error
    error_message = Column(Text, nullable=True)

class ResumeResult(Base):
    __tablename__ = "resume_results"
    id = Column(Integer, primary_key=True, autoincrement=True)
    job_run_id = Column(String)
    filename = Column(String)
    file_path = Column(String)
    rank = Column(Integer)
    score = Column(Float)
    skills_match = Column(Float)
    experience_match = Column(Float)
    education_match = Column(Float)
    missing_skills = Column(Text)   # JSON string
    strengths = Column(Text)        # JSON string
    summary = Column(Text)
    is_shortlisted = Column(Boolean, default=False)

Base.metadata.create_all(bind=engine)

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```

- Progress bar (filled via JS polling /status/{job_id})

- Auto-redirect to /results/{job_id} when status = "done"

```javascript
// Poll every 2 seconds
async function poll() {
  const res = await fetch(`/status/${jobId}`);
  const data = await res.json();
  document.getElementById("progress-bar").style.width = data.progress_percent + "%";
  document.getElementById("message").innerText = data.message;
  if (data.status === "done") window.location.href = `/results/${jobId}`;
  else if (data.status === "error") showError(data.message);
  else setTimeout(poll, 2000);
}
poll();
```
